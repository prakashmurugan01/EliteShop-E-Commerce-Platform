## INSTALLATION STEPS

### 1. CREATE VIRTUAL ENVIRONMENT
python -m venv venv

# On Windows:
venv\Scripts\activate

# On macOS/Linux:
source venv/bin/activate

### 2. INSTALL DEPENDENCIES
pip install -r requirements.txt

### 3. CREATE DJANGO PROJECT & APPS
django-admin startproject ecommerce_core .
python manage.py startapp accounts
python manage.py startapp products
python manage.py startapp cart
python manage.py startapp orders
python manage.py startapp delivery
python manage.py startapp adminpanel

### 4. CREATE DIRECTORIES
mkdir templates
mkdir static
mkdir static/css
mkdir static/js
mkdir static/images
mkdir media
mkdir media/products

### 5. MIGRATIONS
python manage.py makemigrations
python manage.py migrate

### 6. CREATE SUPERUSER
python manage.py createsuperuser
# Enter username, email, password

### 7. COLLECT STATIC FILES
python manage.py collectstatic --noinput

### 8. RUN SERVER
python manage.py runserver

# Visit: http://127.0.0.1:8000/

### settings.py Configuration:

1. Add apps to INSTALLED_APPS:
   - 'accounts'
   - 'products'
   - 'cart'
   - 'orders'
   - 'delivery'
   - 'adminpanel'
   - 'crispy_forms'
   - 'crispy_bootstrap5'

2. Configure MEDIA files:
   MEDIA_URL = '/media/'
   MEDIA_ROOT = BASE_DIR / 'media'

3. Configure STATIC files:
   STATIC_URL = '/static/'
   STATICFILES_DIRS = [BASE_DIR / 'static']

### urls.py Configuration:

from django.conf import settings
from django.conf.urls.static import static

urlpatterns = [
    path('admin/', admin.site.urls),
    path('', include('products.urls')),
    path('accounts/', include('accounts.urls')),
    path('cart/', include('cart.urls')),
    path('orders/', include('orders.urls')),
    path('admin-panel/', include('adminpanel.urls')),
] + static(settings.MEDIA_URL, document_root=settings.MEDIA_ROOT)

## ================================
## DELIVERY TRACKING WORKFLOW
## ================================

1. Customer places order → Status: PLACED
2. Admin confirms order → Status: CONFIRMED
3. Admin marks as packed → Status: PACKED
4. Admin updates to shipped → Status: SHIPPED
5. Admin marks out for delivery → Status: OUT_FOR_DELIVERY
6. Admin marks delivered → Status: DELIVERED

Each status update can include:
- Location info
- Timestamp
- Admin notes

## ================================
## TROUBLESHOOTING
## ================================

### Issue: Static files not loading
Solution: 
  - Run: python manage.py collectstatic
  - Check: STATIC_URL = '/static/'

### Issue: Images not showing
Solution:
  - Check: MEDIA_URL and MEDIA_ROOT in settings
  - Ensure: url pattern includes static/media

### Issue: Login redirects infinitely
Solution:
  - Check: LOGIN_URL = 'login'
  - Verify: Login URL name matches

### Issue: Form CSRF token error
Solution:
  - Add: {% csrf_token %} in forms
  - Check: MIDDLEWARE includes CsrfViewMiddleware
